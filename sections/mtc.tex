\begin{frame}{}
	\begin{definition}[Mini-Transaction]
		\begin{itemize}
			\item Containing $1-2$ read operations and $\le 2$ write operations.
			\item Each write operation is preceded by a read on the same object.
		\end{itemize}
	\end{definition}

	\fig{width = 0.60\textwidth}{figs/mtc-def}

	\begin{definition}[Mini-Transaction History]
		\begin{itemize}
			\item Comprising only mini-transactions;
			\item Satisfying the \textsc{UniqueVal} assumption.
		\end{itemize}
	\end{definition}
\end{frame}
