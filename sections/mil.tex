\begin{frame}{}
	\begin{definition}[Verifying Client Histories]
		A \blue{\it client history} satisfies $\mathcal{M}$
		if it can be \red{$\wrdep$-extended} to a \purple{\it full history} which satisfies $\mathcal{M}$.
	\end{definition}

	\fig{width = 0.60\textwidth}{figs/mil-client-history}

	\begin{theorem}[Complexity of Verifying Client Histories]
		Verifying consistency of \red{client histories}
		with $\ge \rc$ and a fixed number of $\ge 3$ sessions
		is \textsf{NP-complete},
		even \blue{irrespectively} of the complexity of these predicates.
	\end{theorem}
