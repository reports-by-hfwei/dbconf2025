\begin{frame}{}
	\begin{definition}[Write-Specific Serializability (WSS)]
		A schedule of \blue{\it concurrent} transactions
		should produce \purple{\it the same final database state}
		as that produced by a corresponding \blue{\it serial} schedule
		of the same transactions.
	\end{definition}
